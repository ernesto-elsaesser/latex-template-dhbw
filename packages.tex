\usepackage[utf8]{inputenc} % support special characters in .tex files
\usepackage[T1]{fontenc} % support special characters in output files
\usepackage{lmodern} % decent looking font
\usepackage{graphicx} % support for pictures in document
\usepackage[onehalfspacing]{setspace} % support for changing line spacing within the document
\usepackage[margin=2.5cm,foot=1cm]{geometry} % set page margins according to DHBW guidelines

% nicer listings
\usepackage{listings}
\lstset{
  basicstyle=\ttfamily\footnotesize,
  breaklines=true,
  frame=single}

% footnote spacing
\usepackage[perpage,hang]{footmisc}
\setlength{\skip\footins}{10mm} % space between main text and footnote
\setlength{\footnotesep}{5mm} % space between footnotes
\setlength{\footnotemargin}{3mm} % space after footnote number

% paragraph spacing
\setlength{\parindent}{0mm}
\setlength{\parskip}{2mm}

\usepackage[hidelinks]{hyperref} % support cross-referencing
\usepackage{xurl} % support clickable URLs (with line breaks)
